#Regression
n <- 25 # data points
B <- 100 # number of trees
#creating containers to store the RMSPE for each model
RMSPE1 <- c()
RMSPE2 <- c()
RMSPE3 <- c()
RMSPE4 <- c()
for (i in 1:n) {
  ind <- sample(nrow(df1), nrow(df1), replace = TRUE)
  
  train <- df1[ind, ]
  test <- df1[-ind, ]
  
  p = ncol(train)-1
  
  model1 <- rpart(quality~., data =train) # Single Tree
  model2 <- randomForest(quality~., ntree = B, mtry = p, data = train) #Bagged
  model3 <- randomForest(quality~., ntree = B, data = train) # RF
  model4 <- lm(quality ~., data=train)
  
  yhat1 <- predict(model1, test)
  
  yhat2 <- predict(model2, test)
  
  yhat3 <- predict(model3, test)
  
  yhat4 <- predict(model4, test)
  
  RMSPE1[i] <- sqrt(mean((test$quality - yhat1)^2))
  RMSPE2[i] <- sqrt(mean((test$quality - yhat2)^2))
  RMSPE3[i] <- sqrt(mean((test$quality - yhat3)^2))
  RMSPE4[i] <- sqrt(mean((test$quality - yhat4)^2))
}
